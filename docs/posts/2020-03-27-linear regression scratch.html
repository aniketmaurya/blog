Univariate Regression

\(h() = X + b\)

MSE cost function

\((h(x) - y)^2\)

Gradient Descent


repeat {

    Ø = Ø - ∆J(Ø) = Ø - LR*1/m * sum((h(Ø, b) - Y)*X)    
    
    b = b - ∆J(b) =  b - LR*1/m * sum((h(Ø, b) - Y))
}

def mse(y_true, y_pred):
    cost = 0
    m = len(y_pred)
    for i in range(m):
        cost += (y_pred[i] - y_true[i]) ** 2
    return cost/(2*m)

def der_mse(y_true, y_pred):
    der_cost = 0
    m = len(y_pred)
    for i in range(m):
        der_cost += (y_pred[i] - y_true[i])
    return der_cost

def predict(x):
    return w*x + b
# Intialization of variables

m = len(X)
LR = 0.01
w,b =0,0.1

epochs = 10000
# Training

total_cost = []
for i in range(epochs):
    y_pred = []
    epoch_cost = []
    for num, data in enumerate(zip(X, Y)):
        x, y = data
        y_pred = []
        y_pred.append(w*x + b)
        
        
        cost = mse(Y[num:num+1], y_pred)
        epoch_cost.append(cost)
        der_cost = der_mse(Y[num:num+1], y_pred)

        w -= LR * (1/m) * der_cost * x
        b -= LR * (1/m) * der_cost
        
    total_cost.append(np.mean(epoch_cost))
    if i%500==0:
        print(f'epoch:{i}\t\tcost:{cost}')
