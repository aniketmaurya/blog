from threading import Condition, Lock, Thread
import random
found_divisible = False
num = None
cond_var = Condition()
exit_prog = False
def divisible(x):
    return x%5==0

def finder():
    global found_divisible, num, exit_prog
    while not exit_prog:
        x = random.randint(1, 1000)
        if divisible(x):
            cond_var.acquire()        
            num = x
            found_divisible = True
            cond_var.notify()
            cond_var.release()


def printer():
    global num, found_divisible, exit_prog
    while not exit_prog:
        cond_var.acquire()
        while not found_divisible and not exit_prog:
            cond_var.wait()
        print(num)
        found_divisible = False
        cond_var.release()
printerThread = Thread(target=printer)
printerThread.start()

finderThread = Thread(target=finder)
finderThread.start()
